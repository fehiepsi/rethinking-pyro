    def sim_p(G=1.4):
    x123 = torch.rand(3)
    x4 = (G * x123.sum() - x123[1] - x123[2]) / (2 - G)
    z = torch.cat([x123, x4.unsqueeze(0)]).sum()
    p = torch.cat([x123, x4.unsqueeze(0)]) / z
    return {"H": -(p * p.log()).sum(), "p": p}
